/** * Exportação das configurações por ambiente. * O Knex lerá este objeto para decidir qual conexão usar dependendo de onde o app está rodando. */ export default { // Configuração para o ambiente de trabalho da pessoa desenvolvedora (local) development: { ...shared }, // Configuração para o ambiente real (servidor online) production: { ...shared, // Herda todas as configurações do objeto 'shared' // Gerenciamento de conexões simultâneas para suportar muitos usuários pool: { min: 2, // Mantém no mínimo 2 conexões abertas (prontas para uso) max: 20 // Permite abrir no máximo 20 conexões simultâneas para não sobrecarregar o banco }, }, };
